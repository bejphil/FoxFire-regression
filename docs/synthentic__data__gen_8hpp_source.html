163 template<typename T>
164 Vect<T> SyntheticDataGenerator<T>::BuildBeta(const Mat<T>& X,
165  const unsigned int num_observations, const unsigned int num_vars,
166  const unsigned int num_active, const T correlation, const T SNR) {
167 
168  Vect<T> Beta = Vect<T>::Zero(num_vars);
169 
170  std::vector<int> random_subset = choice_vect(num_vars, num_active);
171 
172  for (const auto& elem : random_subset) {
173  Beta[elem] = (std::rand() % 2) ? (1.0) : (-1.0);
174  }
175 
176  Beta = std::sqrt(
177  SNR * static_cast<T>(num_observations) / (X * Beta).squaredNorm())
178  * Beta;
179 
180  return Beta;
